; -- start enable-xdebug --
; The following snippet of code lives in .../docker-resources/enable-xdebug.txt
; Every time a Docker image is built from a Dockerfile, it is moved to
; /tmp/enable-xdebug.txt on the container. The Dockerfile then puts this information
; into the php.ini file.
zend_extension=/xdebug/modules/xdebug.so
xdebug.remote_enable=1
xdebug.remote_connect_back=1
xdebug.remote_port=9000
xdebug.show_local_vars=0
xdebug.var_display_max_data=10000
xdebug.var_display_max_depth=20
xdebug.show_exception_trace=0
; -- end enable-xdebug --
